30       //  this gives you the distance and speed of encoders
31           double encoderSpeed(double targetPosition, double maxSpeed){
32               double AverageEncoderPosition = (RFMotor.getCurrentPosition() + LFMotor.getCurrentPosition() +RBMotor.getCurrentPosition()+ LBMotor.getCurrentPosition())/4;
33               double distance = targetPosition - AverageEncoderPosition;
34               telemetry.addData("Encoder Speed distance",distance);
35               double power = Range.clip(distance/500, -maxSpeed, maxSpeed); // y = m*x + b
36               return power;
37           }
38   
39           public void setTurnPower(double turnPower, double power){
40               RFMotor.setPower(turnPower - power);
41               LFMotor.setPower(-turnPower - power);
42               RBMotor.setPower(turnPower - power);
43               LBMotor.setPower(-turnPower - power);
44           }
45   
46           double getHeading(){
47               angles=imu.getAngularOrientation(AxesReference.INTRINSIC, AxesOrder.ZYX, AngleUnit.DEGREES);
48               this.imu.getPosition();
49               return angles.firstAngle;
50           }
51   
52   
53           double turn(double targetAngle) {
54                   getHeading();
55                   double turnAngle = targetAngle - getHeading();
56                   telemetry.addData("turnAngle", turnAngle);
57                   double power = Range.clip(turnAngle / 50, -0.3, 0.3);
58               return power;
59           }
60   
61           public void rampUp(double distance, double heading, double time, double maxSpeed){
62               double AccelerationSlope = maxSpeed/time;
63               double power = t1.seconds() * AccelerationSlope;
64               if (Math.abs(power) < Math.abs(encoderSpeed(distance, maxSpeed))) { // if acceleration is less than speed
65                   setTurnPower(turn(heading), power);  //then set motor power to turn towards heading and accelerate until max speed
66               }else{
67                       setTurnPower(turn(heading), encoderSpeed(distance, maxSpeed));// otherwise keep motor power to heading and stop at the target Encoder Position
68               }
69           }
